    <script>
        // JavaScript function to change the banner size
        function changeBannerSize() {
            var bannerSize = document.getElementById("bannerSize").value;
            var banner = document.getElementById("banner");
            var bannerImage = document.getElementById("bannerImage");

            switch (bannerSize) {
                case "s":
                    banner.style.width = "1600px";
		    banner.style.height = "900px";
                    break;
                case "m":
		    banner.style.width = "1080px";
                    banner.style.height = "1080px";
                    break;
                case "l":
		    banner.style.width = "1080px";
                    banner.style.height = "1920px";
                    break;
                default:
		     banner.style.width = "1080px";
                    banner.style.height = "1080px"; 
                    break;
            }
        }

        //  change the font size
        function changeFontSize() {
            var fontSize = document.getElementById("fontSize").value;
            document.getElementById("banner").style.fontSize = fontSize + "px";
        }

        //  to change the font color
        function changeFontColor() {
            var fontColor = document.getElementById("fontColor").value;
            document.getElementById("banner").style.color = fontColor;
        }
	 function changeCTAMessage() {
            var ctaMessage = document.getElementById("ctaMessage").value;
            var ctaLinkButton = document.getElementById("ctaLinkButton");
            ctaLinkButton.innerText = ctaMessage;
        }

        // change the CTA link
        function changeCTALink() {
            var ctaLink = document.getElementById("ctaLink").value;
            var ctaLinkButton = document.getElementById("ctaLinkButton");
            ctaLinkButton.href = ctaLink;
        }

        // change the banner text
        function changeBannerText() {
            var newText = prompt("Enter new text for the banner:");
            if (newText !== null) {
                document.getElementById("banner").innerText = newText;
            }
        }
	
	 function changeTextPosition() {
            var xCoordinate = document.getElementById("xCoordinate").value + "px";
            var yCoordinate = document.getElementById("yCoordinate").value + "px";
            var bannerText = document.getElementById("bannerText");
            
            bannerText.style.left = xCoordinate;
            bannerText.style.top = yCoordinate;
        }

	function changeBackgroundColor() {
            var backgroundColor = document.getElementById("backgroundColor").value;
            document.getElementById("banner").style.backgroundColor = backgroundColor;
        }
	
	 function changeBannerImage(event) {
            var imageFile = event.target.files[0];
            var bannerImage = document.getElementById("bannerImage");

            if (imageFile) {
                var imageUrl = URL.createObjectURL(imageFile);
                bannerImage.src = imageUrl;
            } else {
                bannerImage.src = "default-image.jpg"; // Restore the default image if no file is selected
            }
        }
	 

    </script>
